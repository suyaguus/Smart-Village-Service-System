- Langkah instalisasi
├── npx @nestjs/cli new (NAMA-SERVICE) --package-manager pnpm --skip-git
├── pnpm add @prisma/client@7.7.0 @prisma/adapter-pg@7.7.0 pg dotenv @nestjs/mapped-types class-validator class-transformer
├── pnpm add -D prisma@7.7.0 @types/pg
├── pnpm dlx prisma init
├── editing schema prisma
├── buat database di pgAdmin
├── npx prisma generate
├── ubah gitignore menjadi /src/generated/prisma
├── npx prisma db push
├── pnpm dlx @nestjs/cli generate resource (NAMA-SERVICE) (REST API / CRUD[YES])
